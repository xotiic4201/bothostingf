How to Connect a Laptop

                

                    
                

            

            

                
 Step-by-Step Instructions:

                
                

                    
1. Prepare the Laptop

                    
    
                        
  • Install Python 3.8 or higher
    • 
                        
  • Make sure you have pip (Python package manager)
    • 
                    

                

                
                

                    
2. Install Required Packages

                    

                        pip install python-socketio aiohttp psutil
                    

                    
Run this command in your terminal/command prompt

                

                
                

                    
3. Create the Client Script

                    
Create a file named laptop_client.py with the following content:

                    

                        import socketio

                        import aiohttp

                        import asyncio

                        import psutil

                        import platform

                        import socket

                        import uuid

                        import os

                        from datetime import datetime

                        

                        SERVER_URL = "YOUR_BACKEND_URL"  # e.g., "http://localhost:8000" or your deployed URL

                        WORKER_NAME = "My-Laptop"  # Change to your preferred name

                        

                        async def main():

                            sio = socketio.AsyncClient()

                            

                            @sio.event

                            async def connect():

                                print(" Connected to server!")

                                

                                # Register as worker

                                await sio.emit('worker_register', {{

                                    'name': WORKER_NAME,

                                    'hostname': socket.gethostname(),

                                    'platform': platform.system(),

                                    'cpu_count': psutil.cpu_count(),

                                    'total_memory': psutil.virtual_memory().total,

                                    'username': os.getlogin()

                                }})

                                

                            @sio.event

                            async def disconnect():

                                print(" Disconnected from server")

                                

                            @sio.on('deploy_bot_to_worker')

                            async def on_deploy_bot(data):

                                print(f" Received bot deployment: {{data.get('name')}}")

                                

                            @sio.on('start_bot_on_worker')

                            async def on_start_bot(data):

                                print(f" Starting bot: {{data.get('bot_id')}}")

                                

                            @sio.on('stop_bot_on_worker')

                            async def on_stop_bot(data):

                                print(f" Stopping bot: {{data.get('bot_id')}}")

                                

                            try:

                                await sio.connect(SERVER_URL, socketio_path='socket.io')

                                

                                # Send heartbeat every 10 seconds

                                while True:

                                    await sio.emit('worker_heartbeat', {{

                                        'name': WORKER_NAME,

                                        'status': 'online',

                                        'stats': {{

                                            'cpu': psutil.cpu_percent(interval=1),

                                            'memory_percent': psutil.virtual_memory().percent,

                                            'running_bots': []

                                        }},

                                        'last_seen': datetime.now().isoformat()

                                    }})

                                    await asyncio.sleep(10)

                                

                            except Exception as e:

                                print(f" Connection failed: {{e}}")

                        

                        if __name__ == "__main__":

                            asyncio.run(main())

                    

                    
Replace YOUR_BACKEND_URL with your actual backend URL

                

                
                

                    
4. Run the Client

                    

                        python laptop_client.py
                    

                    
The laptop will automatically connect and appear in your dashboard

                

                
                

                    
 Tips:

                    
    
                        
  • You can run this on multiple computers
    • 
                        
  • The laptop will automatically reconnect if connection is lost
    • 
                        
  • Bots deployed to laptops will run even when you close the dashboard
    •
